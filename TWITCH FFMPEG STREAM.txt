#! /bin/bash
 
VBR="750k"
FPS="15"
QUAL="ultrafast"
 
TWITCH="rtmp://live-fra.twitch.tv/app"
KEY="live_727318245_O57lsCTVTvvej2Ddu50BG6v2foIXK3"
 
VIDEO_SOURCE="/home/ubuntu/Desktop/video.mp4"
AUDIO_SOURCE="http://stream.zeno.fm/yuzyuq2d2c9uv"
 
ffmpeg -v quiet \
-re -f lavfi -i "movie=filename=$VIDEO_SOURCE:loop=0, setpts=N/(FRAME_RATE*TB)" \
-thread_queue_size 512 -i "$AUDIO_SOURCE" \
-map 0:v:0 -map 1:a:0 \
-map_metadata:g 1:g \
-vcodec libx264 -pix_fmt yuv420p -preset $QUAL -r $FPS -g $(($FPS * 2)) -b:v $VBR \
-acodec libmp3lame -ar 44100 -threads 6 -qscale:v 3 -b:a 320000 -bufsize 750k \
-f flv "$TWITCH/$KEY"
